    args = ['python', str(layer_library_dir / 'test_list_args' / 'layer.py')]

    out_list = subprocess.Popen(args, stdout=PIPE, stderr=PIPE)

    logger.debug(f"In test_layer_cli, stdout:\n{stdout}\nstderr:\n{stderr}")
